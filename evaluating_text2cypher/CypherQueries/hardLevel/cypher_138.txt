MATCH (m:miRNA)-[:causes_or_contributes_to_condition]->(c:Cardiovascular_disease) WHERE ALL(coord IN m.Genomic_coordinates WHERE toLower(coord) STARTS WITH 'chrx' OR toLower(coord) STARTS WITH 'chry') WITH c, COLLECT(DISTINCT {miRNA: m.Label, GenomicCoordinates: m.Genomic_coordinates}) AS miRNAs RETURN c.Label AS CardiovascularDisease, miRNAs ORDER BY CardiovascularDisease
