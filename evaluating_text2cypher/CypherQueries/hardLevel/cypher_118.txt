MATCH (p:pre_miRNA)-[:has_function]->(g:GO) WITH p, g, [base IN ['A','C','G','T'] | SIZE(SPLIT(p.Sequence, base)) - 1] AS base_counts WITH p, g, REDUCE(maxLen = 0, x IN base_counts | CASE WHEN x > maxLen THEN x ELSE maxLen END) AS longest_repeat RETURN p.Label AS pre_miRNA, g.Label AS GO_term, longest_repeat AS Longest_Repetition ORDER BY longest_repeat DESC, p.Label ASC, g.Label ASC LIMIT 1
