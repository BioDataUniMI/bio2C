MATCH (m:miRNA) WITH m, SIZE([x IN SPLIT(m.Sequence, '') WHERE x = 'U']) AS UCount WITH MAX(UCount) AS MaxUCount WITH MaxUCount MATCH (m:miRNA) WITH m, SIZE([x IN SPLIT(m.Sequence, '') WHERE x = 'U']) AS UCount, MaxUCount WHERE UCount = MaxUCount RETURN m.Label AS miRNA, m.Sequence AS Sequence, UCount
