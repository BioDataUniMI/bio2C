MATCH (bp:Biological_process)-[:has_participant]->(m:miRNA)-[:located_in]->(g:GO) WITH bp, g, COUNT(DISTINCT m) AS miRNA_count WITH MIN(miRNA_count) AS MinMiRNACount WITH MinMiRNACount MATCH (bp:Biological_process)-[:has_participant]->(m:miRNA)-[:located_in]->(g:GO) WITH bp, g, COUNT(DISTINCT m) AS miRNA_count, MinMiRNACount WHERE miRNA_count = MinMiRNACount RETURN bp.Label AS BiologicalProcess, g.Label AS LocationGO, miRNA_count AS MinMiRNACount
