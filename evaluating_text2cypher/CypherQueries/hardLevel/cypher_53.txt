MATCH (m:miRNA)-[:participates_in]->(go:GO)-[:part_of]->(bp:Biological_process) WHERE go.Label CONTAINS 'positive' OR go.Label CONTAINS 'negative' WITH go.Label AS GOLabel, bp.Label AS BiologicalProcess, CASE WHEN go.Label CONTAINS 'positive' THEN 'positive' ELSE 'negative' END AS Type, MAX(SIZE(m.Sequence)) AS MaxSeqLength RETURN GOLabel, BiologicalProcess, Type, MaxSeqLength ORDER BY MaxSeqLength DESC, GOLabel ASC LIMIT 4
