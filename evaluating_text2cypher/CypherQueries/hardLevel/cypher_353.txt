MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:participates_in]->(bp:Biological_process) WITH cc, COUNT(DISTINCT m) AS miRNACount WITH MAX(miRNACount) AS maxMiRNACount WITH maxMiRNACount MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:participates_in]->(bp:Biological_process) WITH cc, COUNT(DISTINCT m) AS miRNACount, maxMiRNACount WHERE miRNACount = maxMiRNACount RETURN cc.Label AS CellularComponent, miRNACount AS NumberOfMiRNAs
