MATCH (g:Gene)-[:transcribed_to]->(:pre_miRNA)-[:has_function]->(:Biological_process) WITH DISTINCT g ORDER BY SIZE(g.Sequence) DESC, g.Label ASC LIMIT 100 MATCH (g)-[:transcribed_to]->(:pre_miRNA)-[:has_function]->(bp:Biological_process) WITH bp, COLLECT(DISTINCT g.Label) AS gene_list, COUNT(DISTINCT g) AS gene_count WHERE gene_count = 5 RETURN bp.Label AS biological_process, gene_list
