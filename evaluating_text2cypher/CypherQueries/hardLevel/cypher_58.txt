MATCH (c:Cancer)-[:expresses]->(m:miRNA)-[:under_expressed_in]->(ph:Phenotype) WHERE ph.Label CONTAINS 'Colon' AND (c.Label CONTAINS 'benign' OR c.Label CONTAINS 'malignant' OR c.Description CONTAINS 'benign' OR c.Description CONTAINS 'malignant') AND toUpper(m.Sequence) =~ '.*(A{3}|C{3}|G{3}|U{3}).*' RETURN c.Label AS Cancer, m.Label AS miRNA, m.Sequence AS miRNA_Sequence, ph.Label AS Phenotype
