MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(d1:Disease)-[:has_characteristic]->(p:Phenotype)-[:phenotype_of]->(d2:Disease) WHERE d1 <> d2 WITH DISTINCT g WITH g, SIZE(g.Sequence) AS sequence_length ORDER BY sequence_length DESC, g.Label ASC LIMIT 3 WITH g, ['A','C','G','T'] AS bases UNWIND bases AS b RETURN g.Label AS gene, SUM((SIZE(g.Sequence) - SIZE(REPLACE(g.Sequence, b+b+b+b, ''))) / 4) AS consecutive_4_count
