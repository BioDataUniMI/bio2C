MATCH (m:miRNA)-[:participates_in]->(go:GO)-[:part_of]->(bp:Biological_process) WHERE go.Label CONTAINS 'positive' OR go.Label CONTAINS 'negative' WITH go.Label AS GOLabel, bp.Label AS BiologicalProcess, CASE WHEN go.Label CONTAINS 'positive' THEN 'positive' ELSE 'negative' END AS Type, MIN(SIZE(m.Sequence)) AS MinSeqLength RETURN GOLabel, BiologicalProcess, Type, MinSeqLength ORDER BY MinSeqLength ASC, GOLabel ASC LIMIT 3
