MATCH (m:miRNA) WITH m, (SIZE(m.Sequence) - SIZE(REPLACE(m.Sequence, 'C', ''))) AS cCount WHERE cCount > 0 WITH 1.0 * cCount / SIZE(m.Sequence) AS cPct WITH MIN(cPct) AS minPct MATCH (m:miRNA) WITH m, (SIZE(m.Sequence) - SIZE(REPLACE(m.Sequence, 'C', ''))) AS cCount, minPct WHERE cCount > 0 WITH m, 1.0 * cCount / SIZE(m.Sequence) AS cPct, minPct WHERE cPct = minPct RETURN m.Label AS miRNA, cPct ORDER BY miRNA
