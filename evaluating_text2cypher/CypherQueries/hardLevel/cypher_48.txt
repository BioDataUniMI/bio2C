MATCH (m:miRNA)-[:regulates_activity_of]->(g:Gene) WITH m, g,[coord IN m.Genomic_coordinates | SPLIT(toLower(coord), ':')[0]] AS m_chr, [coord IN g.Genomic_coordinates | SPLIT(toLower(coord), ':')[0]] AS g_chr WITH m, g, [x IN m_chr WHERE x IN g_chr] AS shared WHERE size(shared) > 0 RETURN DISTINCT m.Label AS miRNA, g.Label AS Gene, shared AS Shared_Chromosomes ORDER BY miRNA, Gene LIMIT 20
