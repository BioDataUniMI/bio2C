MATCH (m:miRNA {Label: 'microRNA hsa-mir-374a precursor'}) WITH m, SUBSTRING(m.Sequence, 0, toInteger(SIZE(m.Sequence)/2)) AS first_half, SUBSTRING(m.Sequence, toInteger(SIZE(m.Sequence)/2)) AS second_half RETURN m.Label AS microRNA, SIZE(first_half) - SIZE(REPLACE(first_half,'G','')) AS G_count_first_half, SIZE(second_half) - SIZE(REPLACE(second_half,'G','')) AS G_count_second_half
