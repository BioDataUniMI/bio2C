MATCH (m:miRNA)-[:in_homology_relationship_with]->(pre:pre_miRNA)-[:causes_or_contributes_to_condition]->(d:Disease)-[:disease_causes_disruption_of]->(mf:Molecular_function) WHERE SIZE(m.Sequence) % 2 = 0 AND SIZE(pre.Sequence) % 2 = 0 RETURN m.Labvel AS miRNA, pre.Label AS precursor_miRNA, COLLECT(DISTINCT mf.Description) AS disrupted_molecular_functions
