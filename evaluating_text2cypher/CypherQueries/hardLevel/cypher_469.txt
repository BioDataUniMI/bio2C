MATCH (m:miRNA)-[:regulates_activity_of]->(g:Gene {Label: 'BSG'})-[:causes_or_contributes_to_condition]->(d:Infectious_disease) WITH m, d, COLLECT(DISTINCT g.Label) AS GeneLabels, COUNT(DISTINCT g) AS mediatorGenes WHERE mediatorGenes = 1 RETURN m.Label AS miRNA, SIZE(m.Sequence) AS SequenceLength, GeneLabels, d.Label AS Disease, mediatorGenes
