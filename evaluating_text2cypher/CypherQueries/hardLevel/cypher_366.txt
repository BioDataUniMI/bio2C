MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:part_of]->(g:GO) WITH cc, COUNT(DISTINCT m) AS numMiRNAs WITH MAX(numMiRNAs) AS maxNumMiRNAs WITH maxNumMiRNAs MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:part_of]->(g:GO) WITH cc, COUNT(DISTINCT m) AS numMiRNAs, COLLECT(DISTINCT g.Label) AS GOTerms, maxNumMiRNAs WHERE numMiRNAs = maxNumMiRNAs RETURN cc.Label AS CellularComponent, numMiRNAs AS NumMiRNAs, GOTerms
