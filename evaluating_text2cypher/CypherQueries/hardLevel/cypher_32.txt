MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(a:Autoimmune_disease) WHERE (toLower(g.Sequence) STARTS WITH 'ctc' OR toLower(g.Sequence) STARTS WITH 'ctg') AND toLower(a.Description) CONTAINS 'intolerance' RETURN DISTINCT g.Label AS Gene, a.Label AS Autoimmune_Disease, a.Description AS Description
