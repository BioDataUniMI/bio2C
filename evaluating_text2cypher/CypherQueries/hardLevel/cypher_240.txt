MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component) WITH cc, COLLECT(DISTINCT m) AS mirnas, COUNT(DISTINCT m) AS miRNA_count ORDER BY miRNA_count DESC, cc.Label ASC LIMIT 3 UNWIND mirnas AS miRNA WITH cc, miRNA ORDER BY cc.Label ASC, SIZE(miRNA.Sequence) ASC WITH cc, COLLECT(miRNA)[0] AS shortest RETURN cc.Label AS CellularComponent, shortest.Label AS miRNA, SIZE(shortest.Sequence) AS SequenceLength
