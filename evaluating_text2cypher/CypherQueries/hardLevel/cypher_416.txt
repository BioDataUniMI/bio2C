MATCH (m:miRNA)-[:acts_upstream_of_or_within]->(b:Biological_process) WHERE b.Label CONTAINS 'leukocyte' WITH m, SIZE([n IN SPLIT(m.Sequence, '') WHERE n = 'G']) AS gCount, SIZE([n IN SPLIT(m.Sequence, '') WHERE n = 'C']) AS cCount WITH m, CASE WHEN (gCount + cCount) = 0 THEN 0.0 ELSE 1.0 * (gCount - cCount) / (gCount + cCount) END AS GC_skew RETURN m.Label AS miRNA, GC_skew ORDER BY GC_skew DESC
