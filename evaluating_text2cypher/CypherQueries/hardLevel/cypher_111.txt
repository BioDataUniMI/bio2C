MATCH (m:pre_miRNA)-[:is_causal_somatic_mutation_in]->(p:Phenotype) WHERE (p:Phenotype OR (p)-[:subclassof*1..]->(:Phenotype)) AND (toLower(p.Label) CONTAINS 'breast carcinoma' OR toLower(p.Label) CONTAINS 'lung adenocarcinoma' OR toLower(p.Label) CONTAINS 'thyroid carcinoma') WITH CASE WHEN toLower(p.Label) CONTAINS 'breast carcinoma' THEN 'Breast carcinoma' WHEN toLower(p.Label) CONTAINS 'lung adenocarcinoma' THEN 'Lung adenocarcinoma' WHEN toLower(p.Label) CONTAINS 'thyroid carcinoma' THEN 'Thyroid carcinoma' END AS Cancer_Type, COUNT(DISTINCT m) AS pre_miRNA_count RETURN Cancer_Type, pre_miRNA_count ORDER BY pre_miRNA_count DESC
