MATCH (m:miRNA)-[:causes_or_contributes_to_condition]->(b:Biological_process) WITH m, SIZE(m.Sequence) AS seq_length WITH COLLECT({m:m, len:seq_length}) AS rows, MAX(seq_length) AS maxLen UNWIND rows AS row WITH row.m AS m, row.len AS seq_length WHERE seq_length = maxLen MATCH (m)-[:causes_or_contributes_to_condition]->(b:Biological_process) RETURN m.Label AS miRNA, SIZE([c IN SPLIT(m.Sequence, '') WHERE c = 'G']) AS G_Count, COLLECT(DISTINCT b.Label) AS BiologicalProcesses ORDER BY miRNA
