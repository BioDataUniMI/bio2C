MATCH (p:pre_miRNA)-[r:under_expressed_in]->(a:Autoimmune_disease) WHERE ANY(src IN r.Source WHERE src = 'miR2Disease') AND SIZE(r.PubMedID) > 2 RETURN a.Label AS AutoimmuneDisease, COLLECT(DISTINCT p.Label) AS Pre_miRNAs, COLLECT(DISTINCT r.Source) AS Sources, COLLECT(DISTINCT r.PubMedID) AS PubMedIDs
