MATCH (pre:pre_miRNA)-[:transcribed_from]->(g:Gene)-[:causes_or_contributes_to_condition]->(c:Cancer {Label: 'renal cell adenocarcinoma'}) WITH g, c, SIZE(g.Sequence) AS geneLength ORDER BY geneLength DESC, g.Label ASC LIMIT 3 RETURN g.Label AS Gene, geneLength AS MaxGeneLength, COLLECT(DISTINCT c.Label) AS Cancers
