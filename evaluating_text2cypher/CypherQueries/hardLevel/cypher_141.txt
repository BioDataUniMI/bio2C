MATCH (m:miRNA)-[:regulates_activity_of]->(g:Gene)-[:causes_or_contributes_to_condition]->(d:Disease) WITH m, d, COLLECT(DISTINCT g.Label) AS GeneLabels, COUNT(DISTINCT g) AS mediatorGenes WHERE mediatorGenes = 11 RETURN m.Label AS miRNA, SIZE(m.Sequence) AS SequenceLength, GeneLabels, d.Label AS Disease, mediatorGenes
