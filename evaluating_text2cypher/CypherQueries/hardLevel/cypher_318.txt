MATCH (m:miRNA)-[:regulates_activity_of]->(g1:Gene)-[:genetically_interacts_with]->(g2:Gene) WITH m, g1, g2, SIZE(g1.Sequence) AS g1Len, SIZE(g2.Sequence) AS g2Len WITH MAX(g1Len) AS max_g1Len, MIN(g2Len) AS min_g2Len WITH max_g1Len, min_g2Len MATCH (m:miRNA)-[:regulates_activity_of]->(g1:Gene)-[:genetically_interacts_with]->(g2:Gene) WITH m, g1, g2, SIZE(g1.Sequence) AS g1Len, SIZE(g2.Sequence) AS g2Len, max_g1Len, min_g2Len WHERE g1Len = max_g1Len AND g2Len = min_g2Len RETURN m.Label AS miRNA, g1.Label AS RegulatedGene, g1Len AS RegulatedGeneLength, g2.Label AS InteractingGene, g2Len AS InteractingGeneLength
