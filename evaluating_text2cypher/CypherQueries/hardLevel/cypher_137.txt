MATCH (m:miRNA)-[:causes_or_contributes_to_condition]->(c:Cardiovascular_disease) WHERE NOT c.Label CONTAINS 'syndrome' AND NOT c.Label CONTAINS 'disease' AND NOT c.Label CONTAINS 'disorder' WITH c, COLLECT(DISTINCT m.Label) AS Contributing_miRNAs WHERE SIZE(Contributing_miRNAs) <= 3 RETURN c.Label AS CardiovascularDisease, Contributing_miRNAs
