MATCH (m:miRNA)-[:under_expressed_in]->(c:Cardiovascular_disease) WITH m, c, SIZE(m.Sequence) AS seq_length ORDER BY seq_length ASC WITH COLLECT({miRNA: m.Label, Sequence_length: seq_length, Genomic_coordinates: m.Genomic_coordinates}) AS miRNAs WITH miRNAs[0] AS shortest UNWIND shortest.Genomic_coordinates AS coord RETURN DISTINCT coord AS Chromosome, shortest.miRNA AS miRNA, shortest.Sequence_length AS Sequence_length
