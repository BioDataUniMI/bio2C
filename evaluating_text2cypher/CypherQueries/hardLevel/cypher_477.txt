MATCH (m:miRNA)-[r:over_expressed_in|under_expressed_in]->(d:Disease) WITH CASE WHEN TYPE(r) = 'over_expressed_in' THEN 'over_expressed' ELSE 'under_expressed' END AS expressionType, [x IN labels(d) WHERE x <> 'Disease'][0] AS diseaseType, COUNT(DISTINCT m) AS miRNACount RETURN diseaseType, expressionType, miRNACount ORDER BY diseaseType, expressionType
