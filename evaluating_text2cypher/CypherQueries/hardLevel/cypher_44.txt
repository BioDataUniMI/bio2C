MATCH (m:miRNA)-[:transcribed_from]->(s:Gene), (m)-[:regulates_activity_of]->(t:Gene) WITH m, s, t, SIZE(t.Sequence) - SIZE(s.Sequence) AS Target_Source_Diff, SIZE(s.Sequence) - SIZE(m.Sequence) AS Source_miRNA_Diff RETURN t.Label AS TargetGene, m.Label AS miRNA, s.Label AS SourceGene, Target_Source_Diff, Source_miRNA_Diff
