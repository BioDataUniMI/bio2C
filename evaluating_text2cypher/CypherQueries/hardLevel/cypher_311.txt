MATCH (d:Disease) WHERE d.Label CONTAINS 'malignant colon neoplasm' CALL { WITH d MATCH (m:miRNA)-[:regulates_activity_of]->(:Gene)-[:causes_or_contributes_to_condition]->(d) RETURN COUNT(DISTINCT m) AS Num_miRNAs } CALL { WITH d MATCH (d)-[:has_phenotype]->(p:Phenotype) RETURN COUNT(DISTINCT p) AS Num_Phenotypes } RETURN d.Label AS Disease, Num_miRNAs, Num_Phenotypes
