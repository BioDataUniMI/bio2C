MATCH (m:miRNA {Label: 'microRNA hsa-mir-374a precursor'}) WITH m, m.Sequence AS seq, toInteger(SIZE(m.Sequence)/2) AS halfLen WITH m, SUBSTRING(seq, 0, halfLen) AS first_half, SUBSTRING(seq, halfLen, SIZE(seq) - halfLen) AS second_half RETURN m.Label AS microRNA, (SIZE(first_half) - SIZE(REPLACE(first_half, 'UA', ''))) / 2 AS UA_count_first_half, (SIZE(second_half) - SIZE(REPLACE(second_half, 'UA', ''))) / 2 AS UA_count_second_half
