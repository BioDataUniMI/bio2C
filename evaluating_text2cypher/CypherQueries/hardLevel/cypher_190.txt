MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(d1:Disease)-[:has_characteristic]->(p:Phenotype)-[:phenotype_of]->(d2:Disease) WHERE d1 <> d2 WITH g, SIZE(g.Sequence) AS sequence_length ORDER BY sequence_length DESC, g.Label ASC LIMIT 1 RETURN g.Label AS gene, SUBSTRING(g.Sequence, sequence_length - 30, 30) AS last_30_characters
