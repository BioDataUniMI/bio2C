MATCH (c:Cancer)-[:has_phenotype]->(p:Phenotype)-[:phenotype_of]->(nd:Neurodegenerative_disease) WITH c, COUNT(DISTINCT p) AS numPhenotypes WITH MAX(numPhenotypes) AS maxNumPhenotypes WITH maxNumPhenotypes MATCH (c:Cancer)-[:has_phenotype]->(p:Phenotype)-[:phenotype_of]->(nd:Neurodegenerative_disease) WITH c, COUNT(DISTINCT p) AS numPhenotypes, COLLECT(DISTINCT nd.Label) AS NeurodegenerativeDiseases, maxNumPhenotypes WHERE numPhenotypes = maxNumPhenotypes RETURN c.Label AS Cancer, numPhenotypes AS NumSharedPhenotypes, NeurodegenerativeDiseases
