MATCH (m:miRNA)-[:develops_from]->(p:pre_miRNA) WITH m, p, [coord IN m.Genomic_coordinates | SPLIT(coord, ':')[0]] AS m_chroms, [coord IN p.Genomic_coordinates | SPLIT(coord, ':')[0]] AS p_chroms WITH m, p, [c IN m_chroms WHERE c IN p_chroms] AS common_chroms ORDER BY SIZE(common_chroms) DESC, m.Label ASC LIMIT 3 RETURN m.Label AS miRNA, p.Label AS pre_miRNA, common_chroms, SIZE(common_chroms) AS num_common_chroms
