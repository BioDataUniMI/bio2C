MATCH (nd:Neurodegenerative_disease)-[:subclassof]->(d:Disease)-[:disease_has_feature]->(p:Phenotype)-[:phenotype_of]->(ad:Autoimmune_disease) WHERE nd.Label CONTAINS 'acute' RETURN nd.Label AS NeurodegenerativeDisease, d.Label AS ParentDisease, COUNT(DISTINCT p) AS PhenotypeCount, COLLECT(DISTINCT p.Label) AS Phenotypes, COLLECT(DISTINCT ad.Label) AS RelatedAutoimmuneDiseases ORDER BY PhenotypeCount DESC
