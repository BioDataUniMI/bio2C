MATCH (m:miRNA)-[:regulates_activity_of]->(g1:Gene)-[r:genetically_interacts_with]->(g2:Gene)-[:causes_or_contributes_to_condition]->(d:Disease) WHERE d.Label CONTAINS 'hereditary' RETURN m.Label AS miRNA, g1.Label AS Regulated_Gene, g2.Label AS Interacting_Gene, d.Label AS Disease, r.GeneMANIA_weight
