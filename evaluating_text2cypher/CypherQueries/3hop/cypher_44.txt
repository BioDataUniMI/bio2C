MATCH (bp2:Biological_process), (bp1:Biological_process)-[:part_of]->(bp2)<-[:has_function]-(pre:pre_miRNA)<-[:transcribed_to]-(g:Gene) WHERE bp2.Label CONTAINS 'development' AND NOT bp2.Label CONTAINS 'heart' RETURN bp1.Label AS biological_process, pre.Label AS precursor_miRNA, g.Label AS gene, bp2.Label AS parent_biological_process
