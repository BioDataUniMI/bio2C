MATCH (d1:Disease)-[:predisposes_towards]->(d2:Disease)-[:has_phenotype]->(p:Phenotype {Label:'Autosomal recessive inheritance'})-[:phenotype_of]->(cv:Cardiovascular_disease) WHERE cv.Label CONTAINS 'infant' AND (d1.Label CONTAINS '1') RETURN d1.Label AS PredisposingDisease, d2.Label AS TargetDisease, cv.Label AS CardiovascularDisease
