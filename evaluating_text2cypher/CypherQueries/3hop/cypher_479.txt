MATCH (pre:pre_miRNA)-[:develops_into]->(m:miRNA)-[:regulates_activity_of]->(g:Gene{Label:'SOX2'})-[:causes_or_contributes_to_condition]->(c:Cancer) WHERE (c.Label CONTAINS 'small' OR c.Label CONTAINS 'non-small')  AND m.Label =~ '.*\d$' RETURN pre.Label AS Pre_miRNA, m.Label AS miRNA, g.Label AS Gene, c.Label AS Cancer
