MATCH (m:miRNA)-[:expressed_in]->(d:Disease)-[:subclassof]->(nd:Neurodegenerative_disease)-[:has_phenotype]->(p:Phenotype) WHERE m.Label ENDS WITH '5p' AND nd.Label CONTAINS 'aphasia' AND NOT p.Label =~ '(?i).*(behavior|recognition|brain|thought).*' RETURN m.Label AS miRNA, d.Label AS Disease, nd.Label AS NeurodegenerativeDisease, p.Label AS Phenotype
