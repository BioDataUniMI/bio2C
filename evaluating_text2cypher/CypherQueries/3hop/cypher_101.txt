MATCH (premirna:pre_miRNA)-[:develops_into]->(mirna:miRNA)-[:regulates_activity_of]->(g:Gene)-[:causes_or_contributes_to_condition]->(nd:Neurodegenerative_disease) WHERE ANY(s IN g.Synonym WHERE s CONTAINS 'son of') RETURN g.Label AS Gene, g.Synonym AS GeneSynonyms, mirna.Label AS miRNA, premirna.Label AS Pre_miRNA, nd.Label AS NeurodegenerativeDisease
