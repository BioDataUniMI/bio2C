MATCH (c:Cancer)-[:expresses]->(m:miRNA)-[:regulates_activity_of]->(g:Gene)-[:causes_or_contributes_to_condition]->(d:Disease) WHERE c.Label CONTAINS 'lipomatosis' AND g.Label STARTS WITH 'CC' AND SIZE(g.Label) = 5 AND g.Label ENDS WITH '1' RETURN c.Label AS Cancer, m.Label AS miRNA, g.Label AS Gene, d.Label AS Disease
