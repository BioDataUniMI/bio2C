MATCH (d:Disease)<-[:causes_or_contributes_to_condition]-(g:Gene)-[:transcribed_from]-(p:pre_miRNA), (g)-[:causes_or_contributes_to_condition]->(c:Cancer) WHERE c.Label CONTAINS 'malign' RETURN d.Label AS Disease, labels(d) AS DiseaseType, p.Label AS Precursor_miRNA, g.Label AS Gene, c.Label AS MalignCancer
