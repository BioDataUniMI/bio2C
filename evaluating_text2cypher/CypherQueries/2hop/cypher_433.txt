MATCH (d:Disease)-[:has_phenotype]->(p:Phenotype)<-[:has_phenotype]-(c:Cancer) WHERE (toLower(d.Label) CONTAINS 'brain' OR toLower(d.Description) CONTAINS 'brain') AND (toLower(c.Label) CONTAINS 'breast' OR toLower(c.Description) CONTAINS 'breast') RETURN p.Label AS Phenotype, COUNT(DISTINCT d) AS numDiseases, COUNT(DISTINCT c) AS numCancers, (COUNT(DISTINCT d) + COUNT(DISTINCT c)) AS TotalShared ORDER BY TotalShared DESC, Phenotype ASC LIMIT 5
