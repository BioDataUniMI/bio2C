MATCH (p:pre_miRNA)-[:transcribed_from]->(g:Gene)-[:causes_or_contributes_to_condition]->(d) WHERE (d:Disease OR d:Cancer OR d:Autoimmune_disease OR d:Neurodegenerative_disease OR d:Infectious_disease OR d:Cardiovascular_disease) AND d.Label CONTAINS 'malign' RETURN d.Label AS Disease, labels(d) AS DiseaseLabels, g.Label AS Gene, p.Label AS Precursor
