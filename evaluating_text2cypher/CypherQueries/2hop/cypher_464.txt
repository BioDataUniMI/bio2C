MATCH (g1:Gene)-[r1:genetically_interacts_with]->(g2:Gene)-[r2:genetically_interacts_with]->(g3:Gene) WHERE r1.GeneMANIA_weight = 0.003 AND r2.GeneMANIA_weight = 0.003 AND (SIZE(g1.Sequence) > 3900 OR SIZE(g2.Sequence) > 3900 OR SIZE(g3.Sequence) > 3900) AND g1 <> g2 AND g2 <> g3 AND g1 <> g3 AND ID(g1) < ID(g3) RETURN g1.Label AS Gene1, g2.Label AS Gene2, g3.Label AS Gene3, SIZE(g1.Sequence) AS Gene1_Sequence_Length, SIZE(g2.Sequence) AS Gene2_Sequence_Length, SIZE(g3.Sequence) AS Gene3_Sequence_Length
