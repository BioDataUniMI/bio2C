MATCH (c:Cancer)-[:expresses]->(m:miRNA) WITH m, COUNT(DISTINCT c) AS cancer_count WITH MAX(cancer_count) AS max_cancer_count WITH max_cancer_count MATCH (c:Cancer)-[:expresses]->(m:miRNA) WITH m, COUNT(DISTINCT c) AS cancer_count, max_cancer_count WHERE cancer_count = max_cancer_count MATCH (m)-[:regulates_activity_of]->(g:Gene) WITH m, COLLECT(DISTINCT g.Label) AS Genes, cancer_count RETURN m.Label AS miRNA, Genes, cancer_count ORDER BY miRNA
