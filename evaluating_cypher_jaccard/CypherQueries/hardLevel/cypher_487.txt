MATCH (m:miRNA)-[r:under_expressed_in]->(d:Disease) WHERE toLower(d.Label) = 'hand, foot and mouth disease' AND ANY(mth IN r.Method WHERE toLower(mth) IN ['northern blotting','qrt-pcr etc']) AND ANY(src IN r.Source WHERE src IN ['miR2Disease','dbDEMC']) RETURN DISTINCT m.Label AS miRNA, d.Label AS Disease, r.Method AS Method, r.Source AS Source
