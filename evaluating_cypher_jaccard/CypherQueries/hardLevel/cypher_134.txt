MATCH (m:miRNA)-[r:causes_or_contributes_to_condition]->(c:Cancer) WHERE toLower(c.Label) CONTAINS 'breast' AND r.RNAsister_score IS NOT NULL WITH MIN(r.RNAsister_score) AS minScore MATCH (m:miRNA)-[r:causes_or_contributes_to_condition]->(c:Cancer) WHERE toLower(c.Label) CONTAINS 'breast' AND r.RNAsister_score = minScore RETURN m.Label AS miRNA, c.Label AS Cancer, r.RNAsister_score AS RNAsister_score ORDER BY Cancer, miRNA
