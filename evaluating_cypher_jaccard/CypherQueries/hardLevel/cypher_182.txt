MATCH (m:miRNA {Label: 'microRNA hsa-mir-374a precursor'}) WITH m, SUBSTRING(m.Sequence, 0, toInteger(SIZE(m.Sequence)/2)) AS first_half, SUBSTRING(m.Sequence, toInteger(SIZE(m.Sequence)/2)) AS second_half RETURN (SIZE(first_half) - SIZE(REPLACE(first_half,'UA',''))) / 2 AS UA_count_first_half, (SIZE(second_half) - SIZE(REPLACE(second_half,'UA',''))) / 2 AS UA_count_second_half
