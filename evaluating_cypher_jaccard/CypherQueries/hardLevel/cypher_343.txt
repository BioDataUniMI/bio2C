MATCH (p:pre_miRNA)-[:develops_into]->(m:miRNA)-[:located_in]->(cc:Cellular_component) WITH p, m, COUNT(DISTINCT cc) AS CellularComponentCount WITH MAX(CellularComponentCount) AS MaxComponentCount WITH MaxComponentCount MATCH (p:pre_miRNA)-[:develops_into]->(m:miRNA)-[:located_in]->(cc:Cellular_component) WITH p, m, COUNT(DISTINCT cc) AS CellularComponentCount, MaxComponentCount WHERE CellularComponentCount = MaxComponentCount RETURN p.Label AS pre_miRNA, m.Label AS miRNA, CellularComponentCount AS MaxComponentCount
