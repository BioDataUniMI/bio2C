MATCH (g1:Gene)-[:genetically_interacts_with]->(g2:Gene) WHERE toLower(g1.Label) STARTS WITH 'ct' AND toLower(g2.Label) STARTS WITH 'ct' AND SIZE(g2.Sequence) > SIZE(g1.Sequence) WITH g1, g2, SIZE(g1.Sequence) AS g1_length, SIZE(g2.Sequence) AS g2_length, (SIZE(g2.Sequence) - SIZE(g1.Sequence)) AS length_difference RETURN DISTINCT g1.Label AS Gene1, g2.Label AS Gene2, g1_length AS Gene1_Length, g2_length AS Gene2_Length, length_difference AS Length_Difference ORDER BY length_difference DESC
