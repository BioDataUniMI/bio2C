MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:part_of]->(g:GO) WHERE cc.Label CONTAINS 'membrane' WITH cc, COUNT(DISTINCT g) AS goCount WITH MAX(goCount) AS maxGOCount WITH maxGOCount MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:part_of]->(g:GO) WHERE cc.Label CONTAINS 'membrane' WITH cc, COUNT(DISTINCT g) AS goCount, maxGOCount WHERE goCount = maxGOCount RETURN cc.Label AS CellularComponent, goCount AS NumberOfGOTerms
