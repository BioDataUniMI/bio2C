MATCH (g:Gene)-[:transcribed_to]->(pre:pre_miRNA)-[:develops_into]->(m:miRNA)-[:causes_or_contributes_to_condition]->(d) WITH g, COLLECT(DISTINCT d) AS diseases, COLLECT(DISTINCT CASE WHEN d:Cancer THEN 'Cancer' WHEN d:Autoimmune_disease THEN 'Autoimmune_disease' WHEN d:Neurodegenerative_disease THEN 'Neurodegenerative_disease' WHEN d:Infectious_disease THEN 'Infectious_disease' WHEN d:Cardiovascular_disease THEN 'Cardiovascular_disease' WHEN d:Disease THEN 'Disease' END) AS cats WITH g, diseases, [c IN cats WHERE c IS NOT NULL] AS cats WHERE SIZE(cats) >= 3 AND ANY(d1 IN diseases WHERE ANY(d2 IN diseases WHERE (d1)-[:subclassof]->(d2))) RETURN COUNT(DISTINCT g) AS GeneCount
