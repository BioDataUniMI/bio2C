MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(p:Phenotype {Label: 'Bipolar affective disorder'}) WHERE ANY(coord IN g.Genomic_coordinates WHERE toLower(coord) CONTAINS 'chr15') WITH g, p, SIZE(g.Sequence) AS sequence_length RETURN g.Label AS Gene, g.Genomic_coordinates AS Genomic_coordinates, sequence_length AS Sequence_Length ORDER BY sequence_length DESC LIMIT 1
