MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(a:Autoimmune_disease) WHERE toLower(a.Label) = 'celiac disease' WITH g, a, (SIZE(SPLIT(g.Sequence, 'GA')) - 1) AS GA_count, (SIZE(SPLIT(g.Sequence, 'CCC')) - 1) AS CCC_count RETURN DISTINCT g.Label AS Gene, GA_count, CCC_count, a.Label AS Autoimmune_Disease ORDER BY GA_count DESC, CCC_count DESC
