MATCH (m:miRNA)-[:under_expressed_in]->(i:Infectious_disease) WITH m, i, SIZE(m.Sequence) AS len WITH COLLECT({miRNA:m.Label, seq:m.Sequence, disease:i.Label, len:len}) AS data, MAX(len) AS max_len UNWIND data AS d WITH d WHERE d.len = max_len WITH d.miRNA AS miRNA, d.disease AS Disease, d.seq AS seq, toInteger(SIZE(d.seq)*0.8) AS start_pos RETURN DISTINCT miRNA, Disease, SUBSTRING(seq, start_pos, SIZE(seq)-start_pos) AS Last_20_percent
