MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(d1:Disease)-[:has_characteristic]->(p:Phenotype)-[:phenotype_of]->(d2:Disease) WHERE d1 <> d2 WITH DISTINCT g WITH g, SIZE(g.Sequence) AS sequence_length ORDER BY sequence_length DESC, g.Label ASC LIMIT 1 WITH g, ['AAAA','CCCC','GGGG','TTTT'] AS sequences UNWIND sequences AS seq WITH g, seq, (SIZE(g.Sequence) - SIZE(REPLACE(g.Sequence, seq, ''))) / 4 AS count ORDER BY count DESC, seq ASC LIMIT 1 RETURN g.Label AS gene, seq AS sequence, count
