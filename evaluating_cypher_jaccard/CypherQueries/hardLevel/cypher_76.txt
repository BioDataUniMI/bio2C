MATCH (p:pre_miRNA)-[:under_expressed_in]->(go:GO) WITH p, go, (SIZE(SPLIT(p.Sequence,'AAA'))-1 + SIZE(SPLIT(p.Sequence,'UUU'))-1 + SIZE(SPLIT(p.Sequence,'GGG'))-1 + SIZE(SPLIT(p.Sequence,'CCC'))-1) AS TotalTripletCount WITH p, TotalTripletCount, COLLECT(DISTINCT go.Label) AS terms UNWIND terms AS term WITH p, TotalTripletCount, term ORDER BY p.ID ASC, term ASC WITH p, TotalTripletCount, COLLECT(term) AS GO_terms ORDER BY TotalTripletCount DESC, p.ID ASC LIMIT 3 RETURN p.ID AS pre_miRNA_ID, TotalTripletCount, GO_terms
