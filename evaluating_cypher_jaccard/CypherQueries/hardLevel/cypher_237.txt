MATCH (m:miRNA)-[:causes_or_contributes_to_condition]->(a:Autoimmune_disease) WHERE a.Label CONTAINS 'multiple sclerosis' WITH a, m, SIZE(m.Sequence) AS SeqLength ORDER BY a.Label, SeqLength ASC WITH a, COLLECT({label:m.Label, seq:m.Sequence, len:SeqLength}) AS Mirnas RETURN a.Label AS AutoimmuneDisease, Mirnas[0].label AS miRNA, Mirnas[0].seq AS Sequence, Mirnas[0].len AS SequenceLength
