MATCH (m:miRNA)-[:transcribed_from]->(g:Gene)-[r:causes_or_contributes_to_condition]->(d:Disease) WITH LABELS(d) AS diseaseTypes, COUNT(r) AS relCount WITH COLLECT({diseaseTypes: diseaseTypes, relCount: relCount}) AS rows, MAX(relCount) AS maxCount UNWIND rows AS row WITH row, maxCount WHERE row.relCount = maxCount RETURN row.diseaseTypes AS DiseaseType, row.relCount AS RelationshipCount
