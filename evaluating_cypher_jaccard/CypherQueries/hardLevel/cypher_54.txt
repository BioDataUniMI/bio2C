MATCH (m:miRNA)-[:participates_in]->(go:GO)-[:part_of]->(bp:Biological_process) WHERE go.Label CONTAINS 'positive' OR go.Label CONTAINS 'cell death' WITH go.Label AS GOLabel, bp.Label AS BiologicalProcess, CASE WHEN go.Label CONTAINS 'cell growth' THEN 'cell growth' ELSE 'cell death' END AS Type, MAX(SIZE(m.Sequence)) AS MaxSeqLength RETURN GOLabel, BiologicalProcess, Type, MaxSeqLength ORDER BY MaxSeqLength DESC
