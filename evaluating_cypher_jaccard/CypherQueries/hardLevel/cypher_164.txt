MATCH (m:miRNA)-[:over_expressed_in]->(bp:Biological_process) WITH m, bp, SIZE(m.Sequence) AS seq_length, SIZE(m.Sequence) - SIZE(REPLACE(m.Sequence, 'A', '')) AS A_count ORDER BY seq_length ASC, A_count DESC WITH COLLECT({miRNA: m, biological_process: bp, seq_length: seq_length, A_count: A_count}) AS data, MIN(seq_length) AS min_length UNWIND [x IN data WHERE x.seq_length = min_length] AS shortest RETURN shortest.miRNA.Label AS miRNA, shortest.biological_process.Label AS biological_process, shortest.seq_length AS sequence_length, shortest.A_count AS A_count
