MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:participates_in]->(bp:Biological_process) WHERE bp.Label CONTAINS 'extracellular' WITH cc, COUNT(DISTINCT m) AS miRNACount WITH MIN(miRNACount) AS minMiRNACount WITH minMiRNACount MATCH (m:miRNA)-[:located_in]->(cc:Cellular_component)-[:participates_in]->(bp:Biological_process) WHERE bp.Label CONTAINS 'extracellular' WITH cc, COUNT(DISTINCT m) AS miRNACount, minMiRNACount WHERE miRNACount = minMiRNACount RETURN cc.Label AS CellularComponent, miRNACount AS NumberOfMiRNAs
