MATCH (m:miRNA)-[:regulates_activity_of]->(g:Gene)-[:causes_or_contributes_to_condition]->(d:Disease) WITH g, d, SIZE(g.Label) AS GeneLabelLength, COUNT(DISTINCT m) AS miRNACount WITH GeneLabelLength, g, d, miRNACount WITH GeneLabelLength, MAX(miRNACount) AS MaxMiRNACount WITH GeneLabelLength, MaxMiRNACount MATCH (m:miRNA)-[:regulates_activity_of]->(g:Gene)-[:causes_or_contributes_to_condition]->(d:Disease) WITH g, d, SIZE(g.Label) AS GeneLabelLength, COUNT(DISTINCT m) AS miRNACount, MaxMiRNACount WHERE miRNACount = MaxMiRNACount RETURN DISTINCT d.Label AS Disease, GeneLabelLength, MaxMiRNACount
