MATCH (p:pre_miRNA)-[:over_expressed_in]->(d:Infectious_disease) WITH p, (SIZE(SPLIT(p.Sequence,'CU'))-1) AS CUCount, COLLECT(DISTINCT d.Label) AS diseases UNWIND diseases AS dis WITH p, CUCount, dis ORDER BY p.ID ASC, dis ASC WITH p, CUCount, COLLECT(dis) AS Infectious_disease_terms ORDER BY CUCount DESC, p.ID ASC LIMIT 3 RETURN p.ID AS pre_miRNA_ID, CUCount, Infectious_disease_terms
