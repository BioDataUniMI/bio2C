MATCH (m:miRNA)-[:transcribed_from]->(s:Gene), (m)-[:regulates_activity_of]->(t:Gene) WITH m, s, t, SIZE(SPLIT(m.Sequence, 'GGGGG')) - 1 AS miRNA_GGGGG_Count, SIZE(SPLIT(s.Sequence, 'GGGGG')) - 1 AS Source_GGGGG_Count, SIZE(SPLIT(t.Sequence, 'GGGGG')) - 1 AS Target_GGGGG_Count RETURN DISTINCT t.Label AS TargetGene, m.Label AS miRNA, s.Label AS SourceGene, miRNA_GGGGG_Count, Source_GGGGG_Count, Target_GGGGG_Count ORDER BY Target_GGGGG_Count DESC, Source_GGGGG_Count DESC
