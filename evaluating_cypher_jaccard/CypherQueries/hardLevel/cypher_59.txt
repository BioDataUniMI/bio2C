MATCH (c:Cancer)-[:expresses]->(m:miRNA)-[:under_expressed_in]->(ph:Phenotype) WHERE ph.Label CONTAINS 'Colon' AND (c.Label CONTAINS 'malignant' OR c.Description CONTAINS 'malignant') WITH ph, c, m, [i IN RANGE(0, SIZE(m.Sequence)-2) WHERE SUBSTRING(m.Sequence, i, 2) = 'GU' | SUBSTRING(m.Sequence, i, 2)] AS GUpairs RETURN ph.Label AS Phenotype, c.Label AS Cancer, m.Label AS miRNA, SIZE(GUpairs) AS GU_Count
