MATCH (c:Cancer {Label:'breast carcinoma'})-[:expresses]->(m:miRNA)-[:under_expressed_in]->(ph:Phenotype) WITH ph, c, m, [i IN RANGE(0, SIZE(m.Sequence)-2) WHERE SUBSTRING(m.Sequence, i, 2) = 'CU' | SUBSTRING(m.Sequence, i, 2)] AS CUpairs WHERE SIZE(CUpairs) = 3 RETURN ph.Label AS Phenotype, m.Label AS miRNA
