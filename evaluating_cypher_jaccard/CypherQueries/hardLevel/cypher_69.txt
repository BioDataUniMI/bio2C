MATCH (pre:pre_miRNA)-[:transcribed_from]->(g:Gene)-[:causes_or_contributes_to_condition]->(c:Cancer) WITH g, SIZE(g.Sequence) AS geneLength, COLLECT(DISTINCT c.Label) AS cancers UNWIND cancers AS cancerLabel WITH g, geneLength, cancerLabel ORDER BY g.Label ASC, cancerLabel ASC WITH g, geneLength, COLLECT(cancerLabel) AS Cancers ORDER BY geneLength DESC, g.Label ASC LIMIT 3 RETURN g.Label AS Gene, geneLength AS GeneLength, Cancers
