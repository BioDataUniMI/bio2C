MATCH (m:pre_miRNA)-[:is_causal_somatic_mutation_in]->(p:Phenotype) WITH m, toLower(p.Label) AS lab UNWIND [CASE WHEN lab CONTAINS 'breast carcinoma' THEN 'Breast carcinoma' END, CASE WHEN lab CONTAINS 'lung adenocarcinoma' THEN 'Lung adenocarcinoma' END, CASE WHEN lab CONTAINS 'thyroid carcinoma' THEN 'Thyroid carcinoma' END] AS Cancer_Type WITH Cancer_Type, m WHERE Cancer_Type IS NOT NULL RETURN Cancer_Type, COUNT(DISTINCT m) AS pre_miRNA_count ORDER BY Cancer_Type
