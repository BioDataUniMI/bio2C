MATCH (m:miRNA)-[:under_expressed_in]->(d:Disease) WITH m, d, [coord IN m.Genomic_coordinates | SPLIT(coord, ':')[0]] AS chroms WHERE 'chrX' IN chroms UNWIND chroms AS c WITH m, d, COLLECT(DISTINCT c) AS distinct_chroms WITH m, d, [c IN distinct_chroms WHERE c <> 'chrX'] AS other_chroms WHERE SIZE(other_chroms) >= 2 RETURN DISTINCT d.Label AS Disease, m.Label AS miRNA, m.Genomic_coordinates AS Genomic_Coordinates
