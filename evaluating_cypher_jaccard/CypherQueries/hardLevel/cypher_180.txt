MATCH (m:miRNA)-[:in_homology_relationship_with]->(pre:pre_miRNA)-[:causes_or_contributes_to_condition]->(d:Disease)-[:disease_causes_disruption_of]->(mf:Molecular_function) WHERE d.Label = 'hemophilia A' RETURN m AS miRNA, SUBSTRING(pre.Sequence, toInteger(SIZE(pre.Sequence) * 0.7), SIZE(pre.Sequence)) AS last_30_percent_pre_miRNA_sequence, COLLECT(DISTINCT mf.Description) AS disrupted_molecular_functions
