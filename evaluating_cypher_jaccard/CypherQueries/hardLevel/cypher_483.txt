MATCH (a:Molecular_function)-[r]->(b:Molecular_function) WHERE TYPE(r) IN ['regulated_by','negatively_regulated_by','directly_positively_regulates','directly_regulated_by','directly_negatively_regulated_by'] WITH CASE WHEN TYPE(r) CONTAINS 'negative' THEN 'negative' WHEN TYPE(r) CONTAINS 'positive' THEN 'positive' ELSE 'unspecified' END AS regulation_effect, TYPE(r) AS relationship_type RETURN regulation_effect, relationship_type, COUNT(*) AS total ORDER BY regulation_effect, total DESC
