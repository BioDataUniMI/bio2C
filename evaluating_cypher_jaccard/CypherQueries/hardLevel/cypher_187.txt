MATCH (pre:pre_miRNA)-[:causes_or_contributes_to_condition]->(d) WITH d, CASE WHEN d:Autoimmune_disease THEN 'Autoimmune' WHEN d:Infectious_disease THEN 'Infectious' WHEN d:Cancer THEN 'Cancer' WHEN d:Neurodegenerative_disease THEN 'Neurodegenerative' WHEN d:Cardiovascular_disease THEN 'Cardiovascular' WHEN d:Mental_disease THEN 'Mental' END AS disease_type WHERE disease_type IS NOT NULL RETURN disease_type, COUNT(DISTINCT d) AS count ORDER BY count DESC, disease_type ASC LIMIT 1
