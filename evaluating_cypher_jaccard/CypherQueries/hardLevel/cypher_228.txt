MATCH (m:miRNA)-[:over_expressed_in]->(b:Biological_process) WHERE SIZE(SPLIT(m.Sequence, 'C')) - 1 = 0 OR SIZE(SPLIT(m.Sequence, 'G')) - 1 = 0 OR SIZE(SPLIT(m.Sequence, 'A')) - 1 = 0 OR SIZE(SPLIT(m.Sequence, 'U')) - 1 = 0 RETURN DISTINCT b.Label AS Biological_Process, m.Label AS miRNA, m.Sequence AS Sequence
