MATCH (m:miRNA) WITH m, SUBSTRING(m.Sequence, 0, SIZE(m.Sequence) / 2) AS firstHalf, SUBSTRING(m.Sequence, SIZE(m.Sequence) / 2) AS secondHalf WITH m, SIZE(firstHalf) - SIZE(REPLACE(firstHalf, 'C', '')) AS cFirst, SIZE(secondHalf) - SIZE(REPLACE(secondHalf, 'C', '')) AS cSecond WHERE cFirst = cSecond AND cFirst > 10 RETURN m.Label AS miRNA, cFirst, cSecond
