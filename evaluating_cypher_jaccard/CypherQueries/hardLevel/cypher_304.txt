MATCH (p:Phenotype)<-[:has_phenotype]-(d:Disease)-[:disease_shares_features_of]->(c:Cancer) WHERE NOT (toLower(d.Label) CONTAINS 'nevus of ota' OR toLower(c.Label) CONTAINS 'nevus of ota') AND NOT toLower(p.Label) CONTAINS 'abnormal ' RETURN p.Label AS Phenotype, d.Label AS Disease, c.Label AS Cancer
