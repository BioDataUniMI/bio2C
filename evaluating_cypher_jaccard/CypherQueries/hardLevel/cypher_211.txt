MATCH (m1:miRNA)-[:genetically_interacts_with]->(m2:miRNA)-[:over_expressed_in]->(cd:Cardiovascular_disease) WHERE SIZE([c IN SPLIT(m1.Sequence, '') WHERE c = 'A']) = SIZE([c IN SPLIT(m2.Sequence, '') WHERE c = 'A']) RETURN m1.Label AS miRNA, m2.Label AS Parent_miRNA, SIZE([c IN SPLIT(m1.Sequence, '') WHERE c = 'A']) AS num_A_miRNA, SIZE([c IN SPLIT(m2.Sequence, '') WHERE c = 'A']) AS num_A_parent, cd.Label AS CardiovascularDisease
