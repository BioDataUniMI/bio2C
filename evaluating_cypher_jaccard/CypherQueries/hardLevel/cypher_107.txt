MATCH (m:pre_miRNA)-[:is_causal_somatic_mutation_in]->(p:Phenotype) WHERE (toLower(p.Label) CONTAINS 'lung' OR toLower(p.Label) CONTAINS 'breast') WITH CASE WHEN toLower(p.Label) CONTAINS 'lung' THEN 'lung' WHEN toLower(p.Label) CONTAINS 'breast' THEN 'breast' END AS Category, COUNT(DISTINCT p) AS Phenotype_count RETURN Category, Phenotype_count
