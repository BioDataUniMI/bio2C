MATCH (m:miRNA)-[:regulates_activity_of]->(g1:Gene)-[:genetically_interacts_with]->(g2:Gene) WITH m, g1, g2, SIZE(g1.Sequence) AS g1Len, SIZE(g2.Sequence) AS g2Len WITH MAX(g1Len) AS maxG1Len, MIN(g2Len) AS minG2Len WITH maxG1Len, minG2Len MATCH (m:miRNA)-[:regulates_activity_of]->(g1:Gene)-[:genetically_interacts_with]->(g2:Gene) WITH m, g1, g2, SIZE(g1.Sequence) AS g1Len, SIZE(g2.Sequence) AS g2Len, maxG1Len, minG2Len WHERE g1Len = maxG1Len AND g2Len = minG2Len RETURN m.Label AS miRNA, SIZE(m.Sequence) AS miRNASequenceLength, g1.Label AS RegulatedGene, g1Len AS RegulatedGeneLength, g2.Label AS InteractingGene, g2Len AS InteractingGeneLength
