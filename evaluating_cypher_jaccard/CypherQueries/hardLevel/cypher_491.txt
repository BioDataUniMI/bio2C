MATCH (p:pre_miRNA)-[r:under_expressed_in]->(c:Cancer) UNWIND r.Source AS src WITH c, COLLECT(DISTINCT src) AS sources WHERE 'PhenomiR' IN sources WITH c.Label AS Cancer, SIZE(sources) AS source_count WITH COLLECT({Cancer: Cancer, source_count: source_count}) AS rows, MAX(source_count) AS max_count UNWIND rows AS row WITH row, max_count WHERE row.source_count = max_count RETURN row.Cancer AS Cancer, row.source_count AS source_count
