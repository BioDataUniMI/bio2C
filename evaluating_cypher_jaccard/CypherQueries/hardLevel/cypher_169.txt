MATCH (m:miRNA)-[:over_expressed_in]->(go1:GO)-[:subclassof]->(go2:GO)-[:positively_regulated_by]->(go3:GO) WHERE go2<>go3 AND SIZE(SPLIT(m.Sequence, 'C')) = 1 OR SIZE(SPLIT(m.Sequence, 'A')) = 1 OR SIZE(SPLIT(m.Sequence, 'G')) = 1 OR  SIZE(SPLIT(m.Sequence, 'U')) = 1 RETURN m.Label AS miRNA, m.Sequence AS Sequence, go1.Label AS go_term, go2.Label AS parent_go_term, go3.Label AS positively_regulating_go
