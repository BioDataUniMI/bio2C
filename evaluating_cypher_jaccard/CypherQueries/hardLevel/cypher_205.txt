MATCH (cc:Cellular_component)-[:participates_in]->(bp:Biological_process)-[:has_participant]->(mirna:miRNA)-[r:under_expressed_in]->(d) WHERE r.FDR IS NOT NULL AND (d:Autoimmune_disease OR d:Neurodegenerative_disease OR d:Infectious_disease OR d:Mental_disease OR d:Cancer) WITH CASE WHEN d:Autoimmune_disease THEN 'Autoimmune' WHEN d:Neurodegenerative_disease THEN 'Neurodegenerative' WHEN d:Infectious_disease THEN 'Infectious' WHEN d:Mental_disease THEN 'Mental' WHEN d:Cancer THEN 'Cancer' END AS disease_type, r.FDR AS fdr RETURN disease_type, AVG(fdr) AS average_FDR
