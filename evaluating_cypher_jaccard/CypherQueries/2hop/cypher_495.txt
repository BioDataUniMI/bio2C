MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(id:Infectious_disease)-[:predisposes_towards]->(:Disease) WHERE NONE(coord IN g.Genomic_coordinates WHERE coord CONTAINS 'chr14' OR coord CONTAINS 'chr4') RETURN g.Label AS Gene, id.Label AS Infectious_Disease, g.Genomic_coordinates AS Genomic_Coordinates
