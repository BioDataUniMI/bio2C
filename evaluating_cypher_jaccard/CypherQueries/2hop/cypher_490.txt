MATCH (pre:pre_miRNA)-[:is_causal_somatic_mutation_in]->(ph:Phenotype)-[:phenotype_of]->(cd:Cardiovascular_disease) WHERE ANY(coord IN pre.Genomic_coordinates WHERE coord ENDS WITH '+') AND ANY(coord IN pre.Genomic_coordinates WHERE coord ENDS WITH '-') RETURN pre.Label AS pre_miRNA, ph.Label AS Phenotype, cd.Label AS Disease, pre.Genomic_coordinates AS Genomic_Coordinates
