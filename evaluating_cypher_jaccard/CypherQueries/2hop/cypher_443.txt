MATCH (c:Cancer)-[:expresses]->(m:miRNA)-[:participates_in]->(bp:Biological_process) WHERE (c.Label CONTAINS 'muscle' OR c.Description CONTAINS 'muscle') AND ANY(coord IN m.Genomic_coordinates WHERE coord CONTAINS 'chr9') AND (bp.Label CONTAINS 'cell cycle' OR bp.Description CONTAINS 'cell cycle') RETURN c.Label AS Cancer, m.Label AS miRNA, bp.Label AS BiologicalProcess
