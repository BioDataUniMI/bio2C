MATCH (pre:pre_miRNA)-[:develops_into]->(m:miRNA)-[:regulates_activity_of]->(g:Gene) WHERE pre.Label CONTAINS '519a-2' AND SIZE(g.Label) = 4 RETURN pre.Label AS Pre_miRNA, m.Label AS Regulator_miRNA, g.Label AS Gene, pre.Genomic_coordinates AS Pre_miRNA_GenomicCoordinates, m.Genomic_coordinates AS miRNA_GenomicCoordinates
