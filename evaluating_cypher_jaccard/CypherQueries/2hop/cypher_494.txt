MATCH (g:Gene)-[:causes_or_contributes_to_condition]->(ad:Autoimmune_disease)-[:predisposes_towards]->(:Disease) WHERE ANY(coord IN g.Genomic_coordinates WHERE coord CONTAINS 'chr3' OR coord CONTAINS 'chr11') RETURN g.Label AS Gene, ad.Label AS Autoimmune_Disease, g.Genomic_coordinates AS Genomic_Coordinates
