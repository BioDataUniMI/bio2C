MATCH (m:miRNA) WHERE m.Sequence STARTS WITH 'A' AND m.Sequence ENDS WITH 'U' AND SIZE(m.Sequence) > 100 AND (ANY(coord IN m.Genomic_coordinates WHERE coord STARTS WITH 'chr1:') OR ANY(coord IN m.Genomic_coordinates WHERE coord STARTS WITH 'chr2:')) RETURN m.Label AS miRNA, m.Sequence AS Sequence, m.Genomic_coordinates AS GenomicCoordinates
