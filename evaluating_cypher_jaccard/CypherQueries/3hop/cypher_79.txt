MATCH (cc:Cellular_component)-[:has_part]->(mirna:miRNA)-[:develops_from]->(premirna:pre_miRNA)-[r:enables]->(mf:Molecular_function) WHERE mf.Label CONTAINS 'binding' RETURN cc.Label AS CellularComponent, mirna.Label AS miRNA, premirna.Label AS Pre_miRNA, mf.Label AS MolecularFunction, r.Source AS Source
