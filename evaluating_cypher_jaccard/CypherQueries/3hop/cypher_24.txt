MATCH (g:Gene {Label:'KL'})-[:causes_or_contributes_to_condition]->(d1:Disease)-[:has_characteristic]->(p:Phenotype)-[:phenotype_of]->(d2:Disease) WHERE d1.Label CONTAINS 'infantile' AND d2.Label CONTAINS 'chronic benign' RETURN d1.Label AS primary_disease, p.Label AS phenotype, d2.Label AS related_disease
