MATCH (cc:Cellular_component)-[:participates_in]->(bp:Biological_process)-[:has_participant]->(mirna:miRNA)-[r:under_expressed_in]->(d:Disease) WHERE d.Label CONTAINS 'noma' AND r.FDR > 0.008 RETURN cc.Label AS CellularComponent, bp.Label AS BiologicalProcess, mirna.Label AS miRNA, d.Label AS Disease, r.FDR AS FDR
