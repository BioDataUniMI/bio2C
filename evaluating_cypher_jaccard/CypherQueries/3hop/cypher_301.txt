MATCH (c:Cancer)-[:expresses]->(m:miRNA)-[:regulates_activity_of]->(g:Gene {Label : 'SYNCRIP'})-[:causes_or_contributes_to_condition]->(d:Disease) WHERE c.Label CONTAINS 'benign' AND NOT d.Label CONTAINS 'carcinoma' AND NOT d.Label CONTAINS 'cancer' AND NOT d.Label CONTAINS 'neoplasm' AND NOT d.Label CONTAINS 'plastic' AND NOT d.Label CONTAINS 'syndrome' RETURN c.Label AS Cancer, m.Label AS miRNA, d.Label AS AssociatedDisease
