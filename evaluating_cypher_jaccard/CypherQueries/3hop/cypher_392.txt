MATCH (p:Phenotype)<-[:has_phenotype]-(cd:Cardiovascular_disease)<-[:causes_or_contributes_to_condition]-(g:Gene)<-[:regulates_activity_of]-(m:miRNA) RETURN g.Label AS Gene, m.Label AS miRNA, cd.Label AS CardiovascularDisease, COUNT(DISTINCT p) AS NumPhenotypes ORDER BY NumPhenotypes DESC, g.Label ASC, m.Label ASC, cd.Label ASC LIMIT 5
